def add_fourier_terms(
    df,
    num,
    seasonal,
    t_start=0,
):
    df_tbl = df.copy()
    t = np.arange(t_start, t_start + len(df_tbl))
    for i in range(1, num + 1):
        df_tbl["sin_"+str(i)] = np.sin(2*np.pi*i*t/seasonal)
        df_tbl["cos_"+str(i)] = np.cos(2*np.pi*i*t/seasonal)

    return df_tbl

from statsmodels.tsa.stattools import adfuller
from statsmodels.tsa.stattools import kpss

def adf_test(
    series, #時系列データ
    reg="c" #"c"（定数項のみ）、"ct"（定数＋トレンド）など指定
):
    result_adf = adfuller(series.dropna(), regression=reg)
    print("===ADF検定結果===")
    print(f"検定統計量(ADF Statistic)：{result_adf[0]:.4f}")
    print(f"p値(p-value)：{result_adf[1]:.4f}")
    if result_adf[1] < 0.05:
        print("帰無仮説を棄却 → 定常\n")
    else:
        print("帰無仮説を棄却できず → 非定常かもしれない\n")

def kpss_test(
    series,
    reg="c"
):
    result_kpss = kpss(series.dropna(), regression=reg)
    print("===KPSS検定結果===")
    print(f"検定統計量(KPSS Statistic)：{result_kpss[0]:.4f}")
    print(f"p値(p-value)：{result_kpss[1]:.4f}")
    if result_kpss[1] < 0.05:
        print("帰無仮説を棄却 → 非定常\n")
    else:
        print("帰無仮説を棄却できず → 定常かもしれない\n")

from statsmodels.graphics.tsaplots import plot_acf
from statsmodels.graphics.tsaplots import plot_pacf
from statsmodels.tsa.seasonal import STL

def plot_and_stationarity_tests(
    series, #時系列データ
    title, #グラフタイトル
    adf_reg="c",
    kpss_reg="c"
):

    # STL分解の実施
    stl = STL(series,
              period=12, #季節周期
             )
    res = stl.fit()
    # 結果をプロット
    fig = res.plot()
    plt.gcf().set_size_inches(10,6)
    plt.tight_layout()
    plt.show()

    # 時系列プロット・コレログラム
    plt.figure(figsize=(10,4))
    # 時系列プロット
    plt.subplot(311)
    plt.plot(series)
    plt.title(title)
    # ACFコレログラム
    plt.subplot(312)
    plot_acf(series.dropna(), ax=plt.gca(), lags=12)
    # PACFコレログラム
    plt.subplot(313)
    plot_pacf(series.dropna(), ax=plt.gca(), lags=12)
    plt.tight_layout()
    plt.show()
    # ADF検定
    adf_test(series, reg=adf_reg)
    # KPSS検定
    kpss_test(series, reg=kpss_reg)

import matplotlib.pyplot as plt
import matplotlib.dates as mdates
from datetime import timedelta

# 予測結果をプロットする関数
def plot_prediction_results(
    y, #実際の観測地
    fitted, #学習データ期間の予測値
    forecast=None, #将来予測の予測値
    conf=None, #将来予測の予測区間
):
    # figとaxオブジェクトを作成
    fig, ax = plt.subplots(figsize=(12, 6))

    # データをプロット (y_plotはインデックスがTimestamp型のSeries)
    ax.plot(y.index, y.values, label="Actual")
    ax.plot(fitted.index, fitted.values, label="Fitted", color = "green", linestyle=":")
    
    
        # 将来の予測値のプロット
    if forecast is not None:
        plt.plot(
            forecast.index,
            forecast,
            label="Forecast",
            color="red",
            linestyle="--"
        )
    if conf is not None:
        plt.fill_between(
            forecast.index,
            conf[:,0], #塗りつぶし幅の下限
            conf[:,1], #塗りつぶし幅の上限
            color="red",
            alpha=0.1, #塗りつぶし幅の透明度
            label="Predicted Interval"
        )

    # ★ 毎月の主目盛り（Major Locator）を設定
    # MonthLocator() は、毎月1日の位置に目盛りを配置します。
    start_date = y.index.min()
    ax.set_xlim(start_date)
    ax.xaxis.set_major_locator(mdates.MonthLocator())
    # ★ 目盛りの表示形式を 'YYYY-MM' に設定
    ax.xaxis.set_major_formatter(mdates.DateFormatter('%Y-%m'))
    # ラベルが重ならないように回転させる
    plt.xticks(rotation=45, ha='right')
    # レイアウトの調整
    plt.tight_layout()
    plt.legend()
    plt.grid(True, linestyle='--', alpha=0.6)
    plt.show()
